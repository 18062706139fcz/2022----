### 经典全排列，变化的是元素而不是坑的个数

```js
function permute = function(nums) {
    let res = []
    let curr = []
    let visited = {}
    const len = nums.length
    function dfs(nth) {
        if(nth === len) {
            return ;
        }
        for(let i=0;i<len;i++) {
            if(!visited[nums[i]]){
                curr.push(nums[i])
                visited[nums[i]] = 1
                dfs(nth+1)
                visited[nums[i]] = 0
                curr.pop()
            }
        }
    }
    dfs(0)
    return res;
}
```


### 类似于全排列，但是变化的是坑的个数，不变的是元素

```js
function subsets = function(nums) {
    let res = []
    let subset = []
    const len = nums.length
    dfs(0)
    function dfs(index) {
        res.push(subset.slice())
        for(let i=index;i<len;i++){
            subset.push(nums[i])
            dfs(i+1)
            subset.pop()
        }
    }
    return res
}
```

### 有特定的数据范围与数据个数的要求「1-n|k」

```js
function subsets = function(n, k) {
    let res = []
    let subset = []
    dfs(1)
    function dfs(index) {
        if(subset.length === k) {
            res,push(subset.slice())
            // return是非常必要的
            return;
        }
        for(let i=index;i<=n;i++){
            subset.push(nums[i])
            dfs(i+1)
            subset.pop()
        }
    }
    return res
}
```
